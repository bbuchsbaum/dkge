```r
test_that("dkge_fit is deterministic given same input data and seed", {
  # Use 5 diverse seeds: 1, 42, 123, 999, 2024
  seeds <- c(1, 42, 123, 999, 2024)

  # Generate fixed test data once
  withr::local_seed(100)
  sim <- dkge_sim_toy(S = 4, P = 50, q = 5, rank = 2, snr = 10)
  data <- dkge_data(sim$betas, sim$designs)
  K <- design_kernel(list(A = list(L = 5, type = "ordinal", l = 1.0)),
                     basis = "cell", normalize = "none")$K

  # Fit with each seed and compare
  fits <- lapply(seeds, function(s) {
    withr::local_seed(s)
    dkge_fit(data, K = K, rank = 2, w_method = "none")
  })

  # All fits should produce identical U (within tolerance)
  for (i in 2:length(fits)) {
    expect_equal(fits[[1]]$U, fits[[i]]$U, tolerance = 1e-10,
                 info = paste("Seed", seeds[i], "differs from seed", seeds[1]))
  }
})
```

```r
test_that("recovery quality is stable across data generation seeds", {
  seeds <- c(1, 42, 123, 999, 2024)
  cosines <- numeric(length(seeds))

  for (i in seq_along(seeds)) {
    withr::local_seed(seeds[i])
    sim <- dkge_sim_toy(S = 5, P = 100, q = 4, rank = 1, snr = 10)
    data <- dkge_data(sim$betas, sim$designs)
    K <- design_kernel(list(A = list(L = 4, type = "ordinal", l = 1.0)),
                       basis = "cell", normalize = "none")$K
    fit <- dkge_fit(data, K = K, rank = 1, w_method = "none")
    cosines[i] <- dkge_cosines_K(fit$U[, 1], sim$U[, 1], K)
  }

  # All cosines should exceed threshold
  expect_true(all(cosines > 0.90),
              info = paste("Cosines:", paste(round(cosines, 3), collapse = ", ")))

  # Variance should be low (< 10% of mean)
  expect_lt(sd(cosines) / mean(cosines), 0.10,
            info = paste("CV:", round(sd(cosines) / mean(cosines), 3)))
})
```

```r
test_that("K-orthonormality holds across seeds", {
  seeds <- c(1, 42, 123, 999, 2024)

  for (seed in seeds) {
    withr::local_seed(seed)
    sim <- dkge_sim_toy(S = 4, P = 50, q = 5, rank = 3, snr = 8)
    data <- dkge_data(sim$betas, sim$designs)
    K <- design_kernel(list(A = list(L = 5, type = "ordinal", l = 1.0)),
                       basis = "cell", normalize = "none")$K
    fit <- dkge_fit(data, K = K, rank = 3, w_method = "none")

    # K-orthonormality: t(U) %*% K %*% U = I
    UtKU <- t(fit$U) %*% K %*% fit$U
    expect_equal(UtKU, diag(ncol(fit$U)), tolerance = 1e-8,
                 info = paste("Seed:", seed))
  }
})

test_that("kernel root reconstruction holds across seeds", {
  seeds <- c(1, 42, 123, 999, 2024)

  for (seed in seeds) {
    withr::local_seed(seed)
    # Generate random PSD kernel
    L <- matrix(rnorm(25), 5, 5)
    K <- crossprod(L)
    K <- (K + t(K)) / 2

    roots <- kernel_roots(K)
    reconstructed <- roots$Khalf %*% roots$Khalf

    expect_equal(reconstructed, K, tolerance = 1e-8,
                 info = paste("Seed:", seed))
  }
})

test_that("Chat symmetry holds across seeds", {
  seeds <- c(1, 42, 123, 999, 2024)

  for (seed in seeds) {
    withr::local_seed(seed)
    sim <- dkge_sim_toy(S = 4, P = 50, q = 4, rank = 2, snr = 10)
    data <- dkge_data(sim$betas, sim$designs)
    K <- design_kernel(list(A = list(L = 4, type = "ordinal", l = 1.0)),
                       basis = "cell", normalize = "none")$K
    fit <- dkge_fit(data, K = K, rank = 2, w_method = "none")

    expect_true(isSymmetric(fit$Chat, tol = 1e-10),
                info = paste("Seed:", seed))
  }
})
```
