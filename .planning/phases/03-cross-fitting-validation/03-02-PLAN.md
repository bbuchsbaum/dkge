**Test 5: Fallback when perturbation magnitude exceeds threshold**
This requires constructing a scenario where the perturbation coefficients are large (> 0.1).
The coefficients are: `coeffs[-j] = -w_s * H[-j, j] / gaps[-j]`

To trigger this:
- Use large subject weight w_s
- Make subject contribution S_s have large off-diagonal couplings in H = V^T S_s V
- Ensure eigenvalue gaps are small (but not small enough to trigger eigengap fallback)

```r
test_that("Analytic LOSO falls back when perturbation magnitude is large", {
  skip_on_cran()

  # Create fit with a dominant subject
  fit <- .make_fit(seed = 106, q = 8, r = 3, S = 4, P = 10)

  # Make subject 1 have very large contribution (large weight * large S_s)
  fit$weights[1] <- 5.0  # Very large weight

  # Scale up the subject's contribution to create large perturbation
  fit$Btil[[1]] <- fit$Btil[[1]] * 10
  fit$contribs[[1]] <- .recompute_contrib(fit$Btil[[1]], fit$Khalf)

  # Recompute Chat to be consistent
  Chat <- matrix(0, nrow(fit$K), ncol(fit$K))
  for (s in seq_along(fit$Btil)) {
    Chat <- Chat + fit$weights[s] * fit$contribs[[s]]
  }
  Chat <- (Chat + t(Chat)) / 2
  fit$Chat <- Chat

  # Recompute full eigen decomposition
  eg <- eigen(Chat, symmetric = TRUE)
  fit$eig_vectors_full <- eg$vectors
  fit$eig_values_full <- eg$values

  cvec <- rnorm(nrow(fit$U))
  result <- dkge_analytic_loso(fit, s = 1, c = cvec, tol = 1e-8)

  # Should fall back due to perturbation magnitude
  expect_equal(result$method, "fallback")
  expect_equal(result$diagnostic$reason, "perturbation_magnitude")
})
```

Add helper at top of file:
```r
.recompute_contrib <- function(Btil_s, Khalf) {
  right <- Btil_s %*% t(Btil_s)
  S_s <- Khalf %*% right %*% Khalf
  (S_s + t(S_s)) / 2
}
```

```r
test_that("Analytic matches iterative LOSO within tolerance for stable input", {
  skip_on_cran()

  # Low-leverage subject = stable regime for perturbation
  fit <- .make_fit(seed = 107, q = 10, r = 3, S = 12, P = 12,
                   low_leverage_subject = 1, low_scale = 1e-4)

  cvec <- rnorm(nrow(fit$U))

  exact <- dkge_loso_contrast(fit, s = 1, c = cvec)
  approx <- dkge_analytic_loso(fit, s = 1, c = cvec, tol = 1e-8)

  expect_equal(approx$method, "analytic")

  # Value comparison: cosine similarity > 0.98
  cosine_v <- sum(approx$v * exact$v) / (sqrt(sum(approx$v^2)) * sqrt(sum(exact$v^2)))
  expect_gt(abs(cosine_v), 0.98)

  # Basis comparison: each column has high cosine with corresponding exact column
  for (j in seq_len(ncol(exact$basis))) {
    cosine_basis <- abs(sum(approx$basis[, j] * exact$basis[, j]))
    expect_gt(cosine_basis, 0.95)  # Slightly lower threshold for basis vectors
  }

  # Eigenvalue comparison: relative error < 1%
  rel_err_lambda <- max(abs(approx$evals - exact$evals[1:length(approx$evals)])) / max(abs(exact$evals[1:length(approx$evals)]))
  expect_lt(rel_err_lambda, 0.01)
})
```
